zds_loaded

struct zds_resbuf *zds_loaded (void);

Determine which ZDS applications are loaded.

This function tells you which ZDS applications are loaded by returning a linked list of all the currently loaded application names. When no applications are loaded, this function returns NULL.

Example

struct zds_resbuf *loadlist;

struct zds_resbuf *templist;

char string1[512];

int howmany = 0;

loadlist=zds_loaded;

for (templist=loadlist;templist->rbnext!=NULL;templist=templist->rbnext){

if (templist->restype!=RTSTR) // It *should* be a string, but if not...

zds_getstring(0,"\nWrong type. Press Enter.",string1);

else{

zds_printf("\nThe application %s is loaded.",templist->resval.rstring);

howmany++;

 }

}

zds_printf("\nThere are %i applications loaded at this time.",howmany);

zds_relrb(loadlist);

loadlist=NULL;

templist=NULL;

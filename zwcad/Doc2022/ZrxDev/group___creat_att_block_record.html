Create a Simple Block Table Record with Attribute Definitions
ZRX Introductory Concepts » Entities » Complex Entity » Create a Simple Block

The following example demonstrates creating a BlockRecord with Attribute and appending it to the Database block table record,then you can use "insert" command to insert this block to the model space.

AcDbObjectId creatAttBlockRecord()
{
AcDbBlockTableRecord *pBlockRecord = new AcDbBlockTableRecord();
pBlockRecord->setName(_T("block_Att"));
AcDbBlockTable *pBlkTbl = NULL;
acdbHostApplicationServices()->workingDatabase()->getBlockTable(pBlkTbl, AcDb::kForWrite);
AcDbObjectId blockTableRecordId;
pBlkTbl->add(blockTableRecordId, pBlockRecord);
pBlkTbl->close();
AcGePoint3d startPt(4.0, 2.0, 0.0);
AcGePoint3d endPt(10.0, 7.0, 0.0);
AcDbLine *pLine = new AcDbLine(startPt, endPt);
AcDbObjectId lineId;
pBlockRecord->appendAcDbEntity(lineId,pLine);
pLine->close();
AcDbAttributeDefinition *pAttdef = new AcDbAttributeDefinition(endPt,_T("line"),_T("Entity type"),_T("input Entity type"));
AcDbObjectId AttdefId;
pBlockRecord->appendAcDbEntity(AttdefId,pAttdef);
pAttdef->close();
pBlockRecord->close();
return blockTableRecordId;
}

Creating Entities
ZRX Introductory Concepts » Database Primer » Creating Objects in ZRX

The following code creates a line.

AcDbObjectId DBPrimer::CreateLine()
{
AcGePoint3d ptStart(0,0,0);
AcGePoint3d ptEnd(100,100,0);
AcDbLine *pLine = new AcDbLine(ptStart,ptEnd);
AcDbBlockTable *pBlockTable;
acdbHostApplicationServices()->workingDatabase()->getBlockTable(pBlockTable,AcDb::kForRead);
AcDbBlockTableRecord *pBlockTableRecord;
pBlockTable->getAt(ACDB_MODEL_SPACE,pBlockTableRecord,AcDb::kForWrite);
AcDbObjectId lineId;
pBlockTableRecord->appendAcDbEntity(lineId,pLine);
pLine->close();
pBlockTableRecord->close();
pBlockTable->close();
return lineId;
}

The following code creates an arc.

AcDbObjectId DBPrimer::CreateArc()
{
AcGePoint3d ptCenter(0,0,0);
AcGeVector3d vec(0,0,1);
double radius = 100;
double startAngle = 10;
double endAngle = 20;
AcDbArc *pArc = new AcDbArc(ptCenter,vec,radius,startAngle,endAngle);
AcDbBlockTable *pBlockTable;
acdbHostApplicationServices()->workingDatabase()->getBlockTable(pBlockTable,AcDb::kForRead);
AcDbBlockTableRecord *pBlockTableRecord;
pBlockTable->getAt(ACDB_MODEL_SPACE,pBlockTableRecord,AcDb::kForWrite);
AcDbObjectId arcId;
pBlockTableRecord->appendZcDbEntity(arcId,pArc);
pArc->close();
pBlockTableRecord->close();
pBlockTable->close();
return arcId;
}

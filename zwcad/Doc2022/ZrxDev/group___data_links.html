Data Links
Advanced Topics » Table Linking

The following example demonstrates to link a table data to an excel spreadsheet.
Warning : there are some related interfaces are not implemented so this feature is temporarily unavailable.

void DataLinkSample()
{
AcDbTable *pTbl = NULL;
AcDbObjectId pEntId;
ads_name en;
ads_point pt;
if (acedEntSel(_T("\nPlease select a table: "), en, pt)
!= RTNORM)
{
acutPrintf(_T("Nothing Selected\n"));
return;
}
acdbGetObjectId(pEntId, en);
acdbOpenObject(pTbl, pEntId, AcDb::kForWrite);
if(NULL == pTbl)
{
acutPrintf(_T("\n Selected entity was not a table!"));
return;
}
struct resbuf *result;
int rc;
if ( (result = acutNewRb(RTSTR)) == NULL)
{
pTbl->close();
acutPrintf(_T("\nError!"));
return;
}
result->resval.rstring=NULL;
rc = acedGetFileD(_T("Excel File"),
_T("c:/test"),
_T("xlsx;xls"),
16,
result);
if (rc != RTNORM)
{
pTbl->close();
acutPrintf(_T("\nError in selecting an EXCEL file!"));
return;
}
ACHAR fileName[MAX_PATH];
_tcscpy(fileName, result->resval.rstring);
rc = acutRelRb(result);
static ACHAR sMyDataLink[MAX_PATH] = _T("DataLinkSample");
Acad::ErrorStatus es;
AcDbDataLinkManager* pDlMan = acdbHostApplicationServices()->workingDatabase()->getDataLinkManager();
assert(pDlMan);
AcDbObjectId idDL;
AcDbDataLink *pDL = NULL;
if( pDlMan->getDataLink(sMyDataLink, pDL, AcDb::kForRead) == Acad::eOk && pDL)
{
pDL->close();
es = pDlMan->removeDataLink(sMyDataLink, idDL);
if( es != Acad::eOk )
{
pTbl->close();
acutPrintf(_T("\nError in removing the Data Link!"));
return;
}
}
es = pDlMan->createDataLink(_T("ZwExcel"), sMyDataLink, _T("data link test."), fileName, idDL);
if( es != Acad::eOk )
{
pTbl->close();
acutPrintf(_T("\nError in creating Data Link!"));
return;
}
es = acdbOpenObject<AcDbDataLink>(pDL, idDL, AcDb::kForWrite);
if ( es != Acad::eOk || !pDL )
{
pTbl->close();
acutPrintf(_T("\nError in opening the Data Link object!"));
return;
}
es = pDL->setOption(AcDb::kDataLinkOptionPersistCache);
es = pDL->setUpdateOption(pDL->updateOption() | AcDb::kUpdateOptionAllowSourceUpdate);
pDL->close();
es = pTbl->setDataLink(1, 0, idDL, true);
if( es != Acad::eOk )
{
pTbl->close();
acutPrintf(_T("\nError in setting Data Link to the selected table!"));
return;
}
es = pTbl->close();
}

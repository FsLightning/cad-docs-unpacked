Getting the entity code information.
This example shows: use the "acedEntSel" function to prompt the user to select a entity and saves the selected result in a "ads_name" variable.
The global function "acdbEntGet" returns a result buffer list that holds the entity defined data.
According to "restype" to determine the current result, the buffer saves the data type, and uses the "GetResvalTypeFromDxftype" function to parse and print the entity's information.
Finally, you need to use the "auctRelRb" function to release the storage space of the result buffer list.

int GetResvalTypeFromDxftype(int restype)
{
int nType=0;
if (restype > -6 && restype < 2000)
{
if (restype < 0)
{
if (restype == -1 || restype == -2)
{
nType = RTENAME;
}
else if (restype == -3)
{
nType = RTNONE;
}
else if (restype == -4)
{
nType = RTSTR;
}
else
{
nType = RTENAME;
}
}
else if (restype < 10)
{
nType = RTSTR;
}
else
{
int nRange=restype/10;
switch(nRange)
{
case 1:
case 11:
case 21:
nType = RT3DPOINT;
break;
case 2:
case 3:
case 4:
case 5:
case 12:
case 13:
case 14:
nType = RTREAL;
break;
case 6:
case 7:
case 17:
case 27:
case 28:
case 29:
case 37:
case 38:
case 40:
nType = RTSHORT;
break;
case 9:
nType = RTLONG;
break;
case 10:
case 30:
case 32:
case 41:
nType = RTSTR;
break;
case 16:
nType = RTINT64;
break;
case 33:
case 34:
case 35:
case 36:
case 39:
nType = RTENAME;
break;
default:
nType = RTERROR;
break;
}
}
}
else
{
nType = RTNONE;
}
return nType;
}
void CSelEntSy::EntityInfo()
{
struct resbuf* rbEnt;
struct resbuf* rb;
ads_name entName;
ads_point point;
if (acedEntSel(_T("Select a Entity: "), entName, point) != RTNORM)
{
return;
}
rbEnt=acdbEntGet(entName);
rb=rbEnt;
int valType=0;
acutPrintf(_T("\n"));
for(rb=rbEnt; rb!=NULL; rb=rb->rbnext)
{
valType = GetResvalTypeFromDxftype(rb->restype);
switch(valType)
{
case RTENAME:
acutPrintf(_T("(%d . <ENTITY NAME: %8lx>) "), rb->restype, rb->resval.rlname[0]);
break;
case RTSHORT:
acutPrintf(_T("(%d . %d) "), rb->restype, rb->resval.rint);
break;
case RTLONG:
acutPrintf(_T("(%d . %ld) "), rb->restype, rb->resval.rlong);
break;
case RTINT64:
acutPrintf(_T("(%d . %I64d) "), rb->restype, rb->resval.mnInt64);
break;
case RTREAL:
acutPrintf(_T("(%d . %0.3f) "), rb->restype, rb->resval.rreal);
break;
case RTSTR:
acutPrintf(_T("(%d . \"%s\") "), rb->restype, rb->resval.rstring);
break;
case RT3DPOINT:
acutPrintf(_T("(%d %0.3f %0.3f %0.3f) "), rb->restype, rb->resval.rpoint[X], rb->resval.rpoint[Y], rb->resval.rpoint[Z]);
break;
}
}
acutPrintf(_T("\n"));
acutRelRb(rbEnt);
}

This example shows: how to use result buffer list to build the entity.
Using the "acutBuildList" function with the group code information to create a list of result buffers which have the same format as returned by "acdbEntGet".
Then creating a new entity based on the content of the result buffer list by using "acdbEntMake" function.

void CSelEntSy::MakeEntity()
{
struct resbuf* rbEnt;
ads_point point1, point2;
ads_real radius=50;
int rc=RTNORM;
point1[X]=100;point1[Y]=100;point1[Z]=0;
point2[X]=20;point2[Y]=50;point2[Z]=0;
rbEnt = acutBuildList(RTDXF0, _T("Line"), 10, point1, 11, point2, RTNONE);
if (rbEnt == NULL)
{
acedAlert(_T("acutBuildList fail."));
return;
}
rc=acdbEntMake(rbEnt);
acutRelRb(rbEnt);
if (rc != RTNORM)
{
acedAlert(_T("acdbEntMake fail."));
}
}

Coordinate System Transformations
Basic Interaction with ZWCAD » Global Functions for Interacting with ZWCAD

Use the acedTrans() function to convert coordinate points or displacement from one coordinate system to another coordinate system.

void InteractGlobalFunction::TranCoorSystem()
{
ads_point pt;
if(acedGetPoint(NULL,_T("please select a point:\n "),pt)==RTNORM)
{
acutPrintf(_T("Point UCS:(%f,%f,%f)\n"),pt[X],pt[Y],pt[Z]);
ads_point result;
struct resbuf fromrb, torb;
fromrb.restype = RTSHORT;
fromrb.resval.rint = 1; // UCS
torb.restype = RTSHORT;
torb.resval.rint = 0; // WCS
acedTrans(pt, &fromrb, &torb, FALSE, result);
acutPrintf(_T("Point WCS:(%f,%f,%f)\n"),result[X],result[Y],result[Z]);
}
}

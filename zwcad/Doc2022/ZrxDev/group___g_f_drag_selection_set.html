Graphically Dragging Selection Sets.

The following code uses the acedDragGen() function to prompt the user to drag a set of selected objects

ads_name ss;
ads_point bp;
void ident_init(ads_matrix id)
{
int i, j;
for (i = 0; i <= 3; i ++)
for (j = 0; j <= 3; j ++)
id[i][j] = 0.0;
for (i = 0; i <= 3; i ++)
id[i][i] = 1.0;
}
void ucs2wcs(ads_point pt)
{
struct resbuf fromrb, torb;
fromrb.restype = RTSHORT;
fromrb.resval.rint = 1; // UCS
torb.restype = RTSHORT;
torb.resval.rint = 0; //WCS
acedTrans(pt, &fromrb, &torb, FALSE, pt);
return;
}
int dragfunc(ads_point usrpt, ads_matrix matrix)
{
ucs2wcs(usrpt);
matrix[0][T] = usrpt[X] - bp[X];
matrix[1][T] = usrpt[Y] - bp[Y];
matrix[2][T] = usrpt[Z] - bp[Z];
return RTNORM;
}
static void MyDragGen(void)
{
ads_point rpt;
acedSSGet(NULL,NULL,NULL,NULL, ss);
acedGetPoint(NULL,_T("\n Select the base point:"),bp);
ucs2wcs(bp);
int rt = acedDragGen(ss, _T("\n Select the insertion point:"), 0, dragfunc, rpt);
if(rt==RTNORM||rt==RTCAN)
{
ucs2wcs(rpt);
ads_matrix mat;
ident_init(mat);
mat[0][T] = rpt[X] - bp[X];
mat[1][T] = rpt[Y] - bp[Y];
mat[2][T] = rpt[Z] - bp[Z];
acedXformSS(ss,mat);
}
acedSSFree(ss);
}

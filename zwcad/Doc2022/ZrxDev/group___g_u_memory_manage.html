Memory Management

Unlike AutoLisp, ZRX does not have automatic garbage collection. In addition to some global functions, in most cases, users need to explicitly free memory.

The following code demonstrates that the iterator must be released manually after use.

void GlobalUtilityFunction::MemoryManage()
{
AcDbBlockTable *pBlockTable;
acdbHostApplicationServices()->workingDatabase()->getBlockTable(pBlockTable,AcDb::kForRead);
AcDbBlockTableRecord *pBlockRecord;
pBlockTable->getAt(ACDB_MODEL_SPACE,pBlockRecord,AcDb::kForRead);
pBlockTable->close();
AcDbBlockTableRecordIterator *pIter;
pBlockRecord->newIterator(pIter);
AcDbEntity *pEnt;
for (pIter->start();!pIter->done();pIter->step())
{
pIter->getEntity(pEnt,AcDb::kForRead);
AcDbHandle entHandle;
pEnt->getZcDbHandle(entHandle);
acutPrintf(_T("%s:%d\n"),pEnt->isA()->name(),entHandle);
pEnt->close();
}
delete pIter;
pBlockRecord->close();
}

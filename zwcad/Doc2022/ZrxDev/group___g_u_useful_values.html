Pause symbol PAUSE consists of a string that contains a backslash
EOS symbol is String termination character

#define PAUSE ZCRX_T("\\")
#define TRUE 1
#define FALSE 0
#define EOS ZCRX_T('\0')

The following code use Pause symbol;

void GlobalUtilityFunction::UsefulValues()
{
ads_name entres;
ads_point ptres;
acedEntSel(NULL, entres, ptres);
acedCommand(RTSTR, _T("rotate"), RTENAME, entres, RTSTR,"", RTSTR, PAUSE, RTREAL, 90.0, 0);
}

Groups and the Group Dictionary
ZRX Introductory Concepts » Container Objects » Dictionaries

The following codes shows how to add objects into group dictionary.

void ContainerObj::AddtoGroup(AcDbObjectIdArray& objIds, ACHAR *groupName)
{
AcDbDictionary *pDictionary;
acdbHostApplicationServices()->workingDatabase()->getGroupDictionary(pDictionary,AcDb::kForRead);
AcDbGroup *pGroup;
pDictionary->getAt(groupName,(AcDbObject* &)pGroup,AcDb::kForWrite);
for(int i=0;i<objIds.length();i++)
{
pGroup->append(objIds[i]);
}
pGroup->close();
pDictionary->close();
}
//The following codes removes a group by name.
void ContainerObj::RemoveGroup()
{
AcDbDictionary *pDictionary;
acdbHostApplicationServices()->workingDatabase()->getGroupDictionary(pDictionary,AcDb::kForWrite);
if(pDictionary->has(_T("TheGroup")))
{
pDictionary->remove(_T("TheGroup"));
}
pDictionary->close();
}

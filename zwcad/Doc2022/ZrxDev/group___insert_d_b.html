Inserting a Database
ZRX Introductory Concepts » Database Operations

Using the insert function of AcDbDatabase to copy all the entities from the model space of a database(Drawing1.dwg) to current database.

void CdatabaseOperations::insertDB()
{
AcDbDatabase *pSourceDb = new AcDbDatabase(Adesk::kFalse);
Acad::ErrorStatus es = Acad::eOk;
if(Acad::eOk != pSourceDb->readDwgFile(_T("Drawing1.dwg")))
{
acedAlert(_T("Readin the file fail."));
return ;
}
if (Acad::eOk != acdbHostApplicationServices()->workingDatabase()->insert(AcGeMatrix3d::kIdentity, pSourceDb))
acedAlert(_T("insert operation fail!"));
delete pSourceDb;
}

Iterating over Tables
ZRX Introductory Concepts » Container Objects » Symbol Tables

The following code iterates through the dimension table.

void ContainerObj::IterateDimStyle()
{
AcDbDimStyleTable *pDimStyleTable;
acdbHostApplicationServices()->workingDatabase()->getSymbolTable(pDimStyleTable,AcDb::kForRead);
AcDbDimStyleTableIterator *pDimIterator;
pDimStyleTable->newIterator(pDimIterator);
AcDbDimStyleTableRecord *pDimStyleTableRecord;
AcDbObjectIdArray objIds;
AcDbObjectId objId;
for(;!pDimIterator->done();pDimIterator->step())
{
pDimIterator->getRecord(pDimStyleTableRecord,AcDb::kForRead);
objId=pDimStyleTableRecord->objectId();
objIds.append(objId);
pDimStyleTableRecord->close();
}
delete pDimIterator;
pDimStyleTable->close();
}

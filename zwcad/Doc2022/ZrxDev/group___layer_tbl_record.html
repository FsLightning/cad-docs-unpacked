Modifying a Layer Table Record
ZRX Introductory Concepts » Container Objects » Symbol Tables

The following codes gets the Layer Table Record from Layer Table by name, then changes the name of the Layer Table Record.

void ContainerObj::ModifyLayerTable( ACHAR *layerName )
{
AcDbLayerTable *pLayerTable;
acdbHostApplicationServices()->workingDatabase()->getLayerTable(pLayerTable,AcDb::kForRead);
if(!pLayerTable->has(layerName))
{
pLayerTable->close();
return;
}
AcDbLayerTableRecord *pLayerTableRecord;
pLayerTable->getAt(layerName,pLayerTableRecord,AcDb::kForWrite);
pLayerTableRecord->setName(_T("NewLayerName"));
pLayerTableRecord->close();
pLayerTable->close();
}

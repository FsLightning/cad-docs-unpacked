ReadIn and Populating a Database
ZRX Introductory Concepts » Database Operations

Using readDwgFile function of AcDbDatabase to load a exiting dwg file into database.
This example shows: load "Drawing1.dwg" file into a new database, then add a poly line with 3 points into database, fanally save the database information back to "Drawing1.dwg" file.

void CdatabaseOperations::readDwgFile()
{
AcDbDatabase *pDb = new AcDbDatabase(Adesk::kFalse);
if(Acad::eOk != pDb->readDwgFile(_T("Drawing1.dwg")))
{
acedAlert(_T("Readin the file fail."));
return ;
}
AcDbBlockTable* pBlockTable=NULL;
AcDbObjectId entId=AcDbObjectId::kNull;
Acad::ErrorStatus es=Acad::eOk;
es=pDb->getBlockTable(pBlockTable, AcDb::kForRead);
if (es==Acad::eOk)
{
AcDbBlockTableRecord* pBlockTableRecord=NULL;
es=pBlockTable->getAt(ACDB_MODEL_SPACE, pBlockTableRecord, AcDb::kForWrite);
pBlockTable->close();
if (es==Acad::eOk)
{
AcGePoint2d pt0(5, 5), pt1(50, 10), pt2(150, 70);
AcDbPolyline* pPLine=new AcDbPolyline(3);
double width = 1;
pPLine->addVertexAt(0, pt0, 0, width, width);
pPLine->addVertexAt(1, pt1, 0, width, width);
pPLine->addVertexAt(2, pt2, 0, width, width);
es=pBlockTableRecord->appendZcDbEntity(entId, pPLine);
if (es==Acad::eOk)
{
pPLine->close();
}
pBlockTableRecord->close();
}
}
es = pDb->saveAs(_T("Drawing1.dwg"));
delete pDb;
}

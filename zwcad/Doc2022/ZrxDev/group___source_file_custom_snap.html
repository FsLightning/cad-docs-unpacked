The following example demonstrates creating a custom osnap mode to capture the one-third the length of a line.
Warning : There are some related interfaces are not implemented so this feature is temporarily unavailable.

ACRX_NO_CONS_DEFINE_MEMBERS(CustomOsnapInfo,AcDbCustomOsnapInfo);
Acad::ErrorStatus CustomOsnapInfo::getOsnapInfo(
AcDbEntity* pickedObject,
Adesk::GsMarker gsSelectionMark,
const AcGePoint3d& pickPoint,
const AcGePoint3d& lastPoint,
const AcGeMatrix3d& viewXform,
AcArray<AcGePoint3d>& snapPoints,
AcDbIntArray & geomIdsForPts,
AcArray<AcGeCurve3d*>& snapCurves,
AcDbIntArray & geomIdsForLines)
{
// Associate with AcDbEntity to define default behavior.
//
snapPoints.setLogicalLength(0);
geomIdsForPts.setLogicalLength(0);
snapCurves.setLogicalLength(0);
geomIdsForLines.setLogicalLength(0);
return Acad::eOk;
}
// Socket Osnap mode protocol extension object for AcDbLine.
//
Acad::ErrorStatus CustomOsnapForLine::getOsnapInfo(
AcDbEntity* pickedObject,
Adesk::GsMarker gsSelectionMark,
const AcGePoint3d& pickPoint,
const AcGePoint3d& lastPoint,
const AcGeMatrix3d& viewXform,
AcArray<AcGePoint3d>& snapPoints,
AcDbIntArray & geomIdsForPts,
AcArray<AcGeCurve3d*>& snapCurves,
AcDbIntArray & geomIdsForLines)
{
ASSERT(pickedObject->isKindOf(AcDbLine::desc()));
AcDbLine* lineEnt = (AcDbLine*)pickedObject;
AcGePoint3d point_1 = AcGePoint3d((lineEnt->endPoint().x + 2*lineEnt->startPoint().x)/3,
(lineEnt->endPoint().y + 2*lineEnt->startPoint().y)/3,0);
snapPoints.append(point_1);
AcGePoint3d point_2 = AcGePoint3d((2*lineEnt->endPoint().x + lineEnt->startPoint().x)/3,
(2*lineEnt->endPoint().y + lineEnt->startPoint().y)/3,0);
snapPoints.append(point_2);//set snap Points.
snapCurves.setLogicalLength(0);
geomIdsForLines.setLogicalLength(0);
return Acad::eOk;
}
Acad::ErrorStatus
CustomOsnapGlyph::setLocation(const AcGePoint3d& dcsPoint)
{
mCurDcsLoc = dcsPoint;
return Acad::eOk;
}
#if _ZWCAD
void
CustomOsnapGlyph::viewportDraw(AcGiViewportDraw* vportDrawContext)
#else
void
CustomOsnapGlyph::subViewportDraw(AcGiViewportDraw* vportDrawContext)
#endif
{
AcGePoint2d pixelArea;
ASSERT(!vportDrawContext->viewport().isPerspective());
vportDrawContext->viewport().getNumPixelsInUnitSquare(AcGePoint3d::kOrigin, pixelArea);
double halfGlyphSizeInDCS = acdbCustomOsnapManager()->osnapGlyphSize()/pixelArea.x ;
AcGePoint3d sSegmentPoints[2];
sSegmentPoints[0].set(
mCurDcsLoc.x-halfGlyphSizeInDCS,
mCurDcsLoc.y-halfGlyphSizeInDCS, 0.0);
sSegmentPoints[1].set(
mCurDcsLoc.x+halfGlyphSizeInDCS,
mCurDcsLoc.y+halfGlyphSizeInDCS, 0.0);
vportDrawContext->geometry().polylineDc(
2, &(sSegmentPoints[0]));
sSegmentPoints[0].set(
mCurDcsLoc.x-halfGlyphSizeInDCS,
mCurDcsLoc.y+halfGlyphSizeInDCS, 0.0);
sSegmentPoints[1].set(
mCurDcsLoc.x+halfGlyphSizeInDCS,
mCurDcsLoc.y-halfGlyphSizeInDCS, 0.0);
vportDrawContext->geometry().polylineDc(
2, &(sSegmentPoints[0]));
sSegmentPoints[0].set(
mCurDcsLoc.x-halfGlyphSizeInDCS,
mCurDcsLoc.y, 0.0);
sSegmentPoints[1].set(
mCurDcsLoc.x+halfGlyphSizeInDCS,
mCurDcsLoc.y, 0.0);
vportDrawContext->geometry().polylineDc(
2, &(sSegmentPoints[0]));
sSegmentPoints[0].set(
mCurDcsLoc.x,
mCurDcsLoc.y-halfGlyphSizeInDCS, 0.0);
sSegmentPoints[1].set(
mCurDcsLoc.x,
mCurDcsLoc.y+halfGlyphSizeInDCS, 0.0);
vportDrawContext->geometry().polylineDc(
2, &(sSegmentPoints[0]));
AcGiTextStyle style;
style.setFileName(TEXT("txt.shx"));
style.setTextSize(halfGlyphSizeInDCS*5);
style.loadStyleRec();
vportDrawContext->geometry().text(mCurDcsLoc, AcGeVector3d::kZAxis,AcGeVector3d::kXAxis ,TEXT("1/3"),3,Adesk::kFalse, style);
}
static CustomOsnapMode* pSocketMode = NULL;
static CustomOsnapInfo* pDefaultSocketInfo = NULL;
static CustomOsnapForLine* pSocketForLine = NULL;
void initApp()
{
// register a command with the ZWCAD command mechanism
// Register the class.
//
CustomOsnapInfo::rxInit();
acrxBuildClassHierarchy();
pDefaultSocketInfo = new CustomOsnapInfo;
AcDbEntity::desc()->addX(CustomOsnapInfo::desc(),
pDefaultSocketInfo);
pSocketForLine = new CustomOsnapForLine;
AcDbLine::desc()->addX(CustomOsnapInfo::desc(),
pSocketForLine);
pSocketGlyph = new CustomOsnapGlyph;
pSocketMode = new CustomOsnapMode;
acdbCustomOsnapManager()->addCustomOsnapMode(pSocketMode);
acdbCustomOsnapManager()->activateOsnapMode(TEXT("line1/3"));
}
void unloadApp()
{
acdbCustomOsnapManager()->removeCustomOsnapMode(pSocketMode);
delete pSocketMode;
// Unregister, then delete the protocol extension object.
//
AcDbEntity::desc()->delX(CustomOsnapInfo::desc());
delete pDefaultSocketInfo;
AcDbLine::desc()->delX(CustomOsnapInfo::desc());
delete pSocketForLine;
// Remove the protocol extension class definition.
//
acrxClassDictionary->remove(TEXT("CustomOsnapInfo"));
}

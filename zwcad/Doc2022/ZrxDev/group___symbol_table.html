Symbol Table
Basic Interaction with ZWCAD » Selection Set, Entity, and Symbol Table Functions

Global function "acdbTblNext" provides a way to obtain symbolic table from the current database.
This example shows: how to use "acdbTblNext" to return the first entry in the "LAYER" table which is specified by the string param.
Then uses the "GetResvalTypeFromDxftype" function to parse and print the layer's information.

void CSelEntSy::TableInfo()
{
struct resbuf* rbEnt;
struct resbuf* rb;
rbEnt = acdbTblNext(_T("LAYER"), 1);
rb=rbEnt;
int valType=0;
acutPrintf(_T("\n"));
for(rb=rbEnt; rb!=NULL; rb=rb->rbnext)
{
valType = GetResvalTypeFromDxftype(rb->restype);
switch(valType)
{
case RTENAME:
acutPrintf(_T("(%d . <ENTITY NAME: %8lx>) "), rb->restype, rb->resval.rlname[0]);
break;
case RTSHORT:
acutPrintf(_T("(%d . %d) "), rb->restype, rb->resval.rint);
break;
case RTLONG:
acutPrintf(_T("(%d . %ld) "), rb->restype, rb->resval.rlong);
break;
case RTINT64:
acutPrintf(_T("(%d . %I64d) "), rb->restype, rb->resval.mnInt64);
break;
case RTREAL:
acutPrintf(_T("(%d . %0.3f) "), rb->restype, rb->resval.rreal);
break;
case RTSTR:
acutPrintf(_T("(%d . \"%s\") "), rb->restype, rb->resval.rstring);
break;
case RT3DPOINT:
acutPrintf(_T("(%d %0.3f %0.3f %0.3f) "), rb->restype, rb->resval.rpoint[X], rb->resval.rpoint[Y], rb->resval.rpoint[Z]);
break;
}
}
acutRelRb(rbEnt);
}

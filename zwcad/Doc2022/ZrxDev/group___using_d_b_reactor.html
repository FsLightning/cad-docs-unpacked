Creating a Database Reactor
Advanced Topics » Notification

The following example creates a reactor derived from AcDbDatabaseReactor to counts the number of objects and print whether the object is a circle when the object was appended.

class DBReactor : public AcDbDatabaseReactor
{
public:
virtual void objectAppended(const AcDbDatabase* dwg, const AcDbObject* dbObj)
{
acutPrintf(_T("\nAcDbDatabaseReactor: objectAppended"));
m_entCount++;
if(dbObj->isKindOf(AcDbCircle::desc()))
{
acutPrintf(_T("\nIt is a circle."));
}
else
{
acutPrintf(_T("\nIt is not a circle."));
}
acutPrintf(_T("\nThe number of entities: %d"),m_entCount);
}
virtual void objectOpenedForModify(const AcDbDatabase* dwg, const AcDbObject* dbObj)
{
acutPrintf(_T("\nAcDbDatabaseReactor: objectOpenedForModify"));
}
virtual void objectErased(const ZcDbDatabase* dwg, const ZcDbObject* dbObj, ZSoft::Boolean bErased)
{
acutPrintf(_T("\nAcDbDatabaseReactor: objectErased"));
if(bErased == Adesk::kTrue)
{
m_entCount--;
acutPrintf(_T("\nThe number of entities: %d"),m_entCount);
}
else
{
m_entCount++;
acutPrintf(_T("\nThe number of entities: %d"),m_entCount);
}
}
static DBReactor *getDBReactor()
{
if(m_dbReactor == NULL)
{
m_dbReactor = new DBReactor();
}
return m_dbReactor;
}
private:
DBReactor(){};
static DBReactor *m_dbReactor;
static int m_entCount;
};
DBReactor * DBReactor::m_dbReactor = NULL;
int DBReactor::m_entCount = 0;

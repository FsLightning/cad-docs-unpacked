Creating an Editor Reactor
Advanced Topics » Notification

The following code creates a reactor derived from AcEditorReactor to print the name of events.

class EditorReactor : public AcEditorReactor
{
public:
virtual void dwgFileOpened(ZcDbDatabase*, ZTCHAR* fileName)
{
acutPrintf(_T("\nAcEditorReactor: dwgFileOpened"));
}
virtual void commandWillStart(const ZTCHAR* cmdStr)
{
acutPrintf(_T("\nAcEditorReactor: commandWillStart(%s)"),cmdStr);
}
virtual void commandEnded(const ZTCHAR* cmdStr)
{
acutPrintf(_T("\nAcEditorReactor: commandEnded(%s)"),cmdStr);
}
static EditorReactor *getEditorReactor()
{
if(m_editorReactor == NULL)
{
m_editorReactor = new EditorReactor();
}
return m_editorReactor;
}
private:
EditorReactor(){};
static EditorReactor *m_editorReactor;
};
EditorReactor * EditorReactor::m_editorReactor = NULL;

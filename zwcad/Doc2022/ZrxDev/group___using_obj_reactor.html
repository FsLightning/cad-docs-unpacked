Creating an Object Reactor
Advanced Topics » Notification

The following code creates a reactor derived from AcDbObjectReactor to print when a object was erased or unerased.

class ObjectReactor : public AcDbObjectReactor
{
public:
virtual void erased(const ZcDbObject* dbObj, ZSoft::Boolean pErasing )
{
if(pErasing == Adesk::kTrue)
{
acutPrintf(_T("\nObjectReactor: erased"));
}
else
{
acutPrintf(_T("\nObjectReactor: unerased"));
}
}
static ObjectReactor *getObjectReactor()
{
if(m_objReactor == NULL)
{
m_objReactor = new ObjectReactor();
}
return m_objReactor;
}
private:
ObjectReactor(){};
static ObjectReactor *m_objReactor;
};
ObjectReactor * ObjectReactor::m_objReactor = NULL;

void
aFunction()
{
acutPrintf(TEXT("This is ZsdkProfileSample Test Application...\n"));
// Attach the reactor for the duration of this command. Normally
// this would be added upon application initialization.
//
// Obtain the path for the registry keys and print it out.
//
ACHAR *pstrKey;
acProfileManagerPtr()->ProfileRegistryKey(pstrKey, NULL);
if (pstrKey != NULL) {
acutPrintf(TEXT("\nThe profiles registry key is: %s"), pstrKey);
acutDelString(pstrKey);
}
// Get the list of all profiles in the user's configuration
// and print them out.
//
AcApProfileNameArray arrNameList;
int nProfiles =
acProfileManagerPtr()->ProfileListNames(arrNameList);
acutPrintf(TEXT("\nNumber of profiles currently in the user profile list is: %d"), nProfiles);
for (int i = 0; i < nProfiles; i++)
acutPrintf(TEXT("\nProfile name is: %s"), arrNameList[i]);
// Copy the unnamed profile to the ZsdkTestProfile.
//
acProfileManagerPtr()->ProfileCopy(
TEXT("ZsdkTestProfile"),
TEXT("Default"),
TEXT("This is a test"));
// Reset the newly copied profile to ZWCAD defaults.
//
acProfileManagerPtr()->ProfileReset(TEXT("ZsdkTestProfile"));
// Make this new profile current.
//
acProfileManagerPtr()->ProfileSetCurrent(TEXT("ZsdkTestProfile"));
// Change a value in the profile. For this example, make the
// cursor big.
//
struct resbuf rbCursorSize;
rbCursorSize.restype = RTSHORT;
rbCursorSize.resval.rint = 100;
acedSetVar(TEXT("CURSORSIZE"), &rbCursorSize);
// Rename the profile to a new name.
//
acProfileManagerPtr()->ProfileRename(
TEXT("ZsdkTestProfile2"),
TEXT("ZsdkTestProfile"),
TEXT("This is another test"));
// Export the profile.
//
acProfileManagerPtr()->ProfileExport(
TEXT("ZsdkTestProfile2"),
TEXT("./ZsdkTestProfile2.arg"));
// Import the profile.
//
acProfileManagerPtr()->ProfileImport(
TEXT("ZsdkTestProfile3"),
TEXT("./ZsdkTestProfile2.arg"),
TEXT("This is a copy of ZsdkTestProfile2 by Exporting/Importing"),
Adesk::kTrue);
//
}

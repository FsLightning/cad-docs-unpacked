Initialize the ZRX application by calling the function:

void initApp()
{
acedRegCmds->addCommand(_T("HELLOWORLD"),
_T("Hello"),
_T("Helo"),
ACRX_CMD_MODAL,
HelloWorld);
}

Unload the ZRX application by calling the function:

void unloadApp()
{
acedRegCmds->removeGroup(_T("HELLOWORLD"));
}

An entrypoint function:

extern "C" AcRx::AppRetCode
acrxEntryPoint(AcRx::AppMsgCode msg, void *p)
{
switch(msg)
{
case AcRx::kInitAppMsg:
acrxDynamicLinker->unlockApplication(p);
acrxRegisterAppMDIAware(p);
initApp();
break;
}
case AcRx::kUnloadAppMsg:
unlocadApp();
break;
default:
break;
}
return AcRx::kRetOK;

A custom function:

void HelloWorld()
{
acutPrintf("Hello World");

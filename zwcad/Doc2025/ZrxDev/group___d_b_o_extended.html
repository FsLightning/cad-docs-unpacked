Extended Data
ZRX Introductory Concepts » Database Objects » Adding Object-Specific Data


The setXData function of the AcDbObject class is used to set the extension data for an object

virtual Acad::ErrorStatus AcDbObject::setXData(const resbuf* xdata);

The xData function of the AcDbObject class is used to get extended data for an object

virtual resbuf* AcDbObject::xData(const char* regappName = NULL) const;

The following example uses the setXdata() function to set the xdata of the selected object.

void DBObjects::AddXData()
{
ads_name en;
ads_point pt;
if(acedEntSel(_T("\nSelect the entity you want to add extended data:"),en,pt) != RTNORM)
{
return;
}
AcDbObjectId entId;
Acad::ErrorStatus es = acdbGetObjectId(entId,en);
struct resbuf* pRb;
acdbRegApp(_T("XData"));
CString strAppName(_T("XData"));;
CString str(_T("hello"));
pRb = acutBuildList(AcDb::kDxfRegAppName,strAppName,
AcDb::kDxfXdAsciiString,str,
AcDb::kDxfXdInteger32,2,
AcDb::kDxfXdReal,3.14,
AcDb::kDxfXdWorldXCoord,pt,
RTNONE);
AcDbEntity* pEnt;
acdbOpenAcDbEntity(pEnt,entId,AcDb::kForWrite);
struct resbuf* pTemp;
pTemp = pEnt->xData(_T("XData"));
if(pTemp != NULL)
{
acutRelRb(pTemp);
acutPrintf(_T("It already contains extended data"));
}
else
{
pEnt->setXData(pRb);
acutPrintf(_T("\n Extended data added successfully"));
}
pEnt->close();
acutRelRb(pRb);
}

The following example uses the xData () function to get the xdata of the selected object, and then print the xdata to the screen

void DBObjects::PrintXData()
{
ads_name en;
ads_point pt;
if(acedEntSel(_T("\nSelect the extended data entity to view:"),en,pt) != RTNORM)
{
return;
}
AcDbObjectId entId;
Acad::ErrorStatus es = acdbGetObjectId(entId,en);
//Open the entity to check if the data is contains an extension
AcDbEntity* pEnt;
acdbOpenAcDbEntity(pEnt,entId,AcDb::kForRead);
struct resbuf* pRb;
pRb = pEnt->xData(_T("XData"));
pEnt->close();
if(pRb != NULL)
{
struct resbuf* pTemp;
pTemp = pRb;
pTemp = pTemp->rbnext;
acutPrintf(_T("\n String type data extension�?s"),pTemp->resval.rstring);
pTemp = pTemp->rbnext;
acutPrintf(_T("\n Integer type data extension�?d"),pTemp->resval.rint);
pTemp = pTemp->rbnext;
acutPrintf(_T("\n Real number type data extension�?.2f"),pTemp->resval.rreal);
pTemp = pTemp->rbnext;
acutPrintf(_T("\n Point type data extension�?.2f,%.2f,%.2f"),pTemp->resval.rpoint[X],pTemp->resval.rpoint[Y],pTemp->resval.rpoint[Z]);
acutRelRb(pRb);
}
else
{
acutPrintf(_T("\n The entity does not contain any extension data"));
}
}

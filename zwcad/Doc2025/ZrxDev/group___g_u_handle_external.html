Handling External Applications
Basic Interaction with ZWCAD » ZRX Global Utility Functions » Common Characteristics of ZRX Library Functions

The following print the names of all currently loaded applications.
And show how to Load or Unload an application:

void GlobalUtilityFunction::HandleExternal()
{
struct resbuf *rb1, *rb2;
for (rb2 = rb1 = acedArxLoaded(); rb2 != NULL; rb2 = rb2->rbnext) {
if (rb2->restype == RTSTR)
acutPrintf(_T("%s\n"), rb2->resval.rstring);
}
if (acedArxUnload(_T("myappname")) != RTERROR) {
acutPrintf(_T("unLoaded successfully\n"));
}
if (acedArxLoad(_T("myappname")) != RTERROR) {
acutPrintf(_T("Loaded successfully\n"));
}
}

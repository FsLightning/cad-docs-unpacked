The following code use acutBuildList and acedCmd function to create an circle.

void GlobalUtilityFunction::UseCmd()
{
struct resbuf *rb;
ads_point ptCenter = {10, 0, 0};
ads_real radius = 10;
rb = acutBuildList(RTSTR,_T("circle"),
RTPOINT, ptCenter,
RTREAL, radius,
RTNONE);
if (rb != NULL)
{
acedCmd(rb);
}
}

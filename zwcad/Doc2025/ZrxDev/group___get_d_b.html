Getting Current Database
ZRX Introductory Concepts » Database Operations

ZRX Application can get the current database pointer by using acdbHostApplicationServices()->workingDatabase().
Through the database pointer, ZRX Application can accesses the block table, linetype table, text style table, layer table, and so on.
This example shows how to add a text to the current database.

void CdatabaseOperations::workingDatabase()
{
AcDbDatabase *pDb = acdbHostApplicationServices()->workingDatabase();
if (pDb != NULL)
{
AcDbBlockTable* pBlockTable=NULL;
AcDbObjectId entId=AcDbObjectId::kNull;
Acad::ErrorStatus es=Acad::eOk;
es=pDb->getBlockTable(pBlockTable, AcDb::kForRead);
if (es==Acad::eOk)
{
AcDbBlockTableRecord* pBlockTableRecord=NULL;
es=pBlockTable->getAt(ACDB_MODEL_SPACE, pBlockTableRecord, AcDb::kForWrite);
pBlockTable->close();
if (es==Acad::eOk)
{
AcDbMText* pText=new AcDbMText();
AcString text=_T("HelloWorld!");
double height=5;
double width=60;
AcGePoint3d position(30, 20, 0);
pText->setContents(text.kTCharPtr());
pText->setLocation(position);
pText->setTextHeight(height);
pText->setWidth(width);
es=pBlockTableRecord->appendZcDbEntity(entId, pText);
if (es==Acad::eOk)
{
pText->close();
}
pBlockTableRecord->close();
}
}
}
}

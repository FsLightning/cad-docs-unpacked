Custom Objects » Deriving from AcDbobject

The following code shows the class declaration for the new class DwgAuthor derived from AcDbObject: A custom class must override the following functions:
virtual Acad::ErrorStatus dxfOutFields(AcDbDxfFiler* pFiler) const;
virtual Acad::ErrorStatus dxfInFields(AcDbDxfFiler* pFiler);
virtual Acad::ErrorStatus dwgOutFields(AcDbDwgFiler* pFiler) const;
virtual Acad::ErrorStatus dwgInFields(AcDbDwgFiler* pFiler);
which are used for persistent storage of objects as well as for copying and undo operations.
GetDwgAuthor and SetDwgAuthor functions are provided for this data member "m_DwgAuthor" for store file Author name.

class DwgAuthor : public AcDbObject
{
public:
ACRX_DECLARE_MEMBERS(DwgAuthor);
DwgAuthor();
virtual ~DwgAuthor();
virtual Acad::ErrorStatus dxfOutFields(AcDbDxfFiler* pFiler) const;
virtual Acad::ErrorStatus dxfInFields(AcDbDxfFiler* pFiler);
virtual Acad::ErrorStatus dwgOutFields(AcDbDwgFiler* pFiler) const;
virtual Acad::ErrorStatus dwgInFields(AcDbDwgFiler* pFiler);
void SetDwgAuthor(AcString DwgAuthor);
AcString GetDwgAuthor() const;
private:
AcString m_DwgAuthor;
};

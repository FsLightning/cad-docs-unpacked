Iterating over Dictionary Entries
ZRX Introductory Concepts » Container Objects » Dictionaries

The following codes uses a iterator to step through the objects in Named Object Dictionary, and then add objectids into a AcDbObjectIdArray.

void ContainerObj::IterDict()
{
AcDbDictionary *pNamedObjDict;
acdbHostApplicationServices()->workingDatabase()->getNamedObjectsDictionary(pNamedObjDict,AcDb::kForRead);
AcDbDictionary *pDict;
pNamedObjDict->getAt(_T("NewDictionary"),(AcDbObject* &)pDict,AcDb::kForRead);
AcDbDictionaryIterator *pDictIter = pDict->newIterator();
AcDbObject *obj;
AcDbObjectIdArray objIds;
for(;!pDictIter->done();pDictIter->next())
{
pDictIter->getObject(obj,AcDb::kForRead);
objIds.append(obj->id());
}
delete pDictIter;
pDict->close();
pNamedObjDict->close();
}

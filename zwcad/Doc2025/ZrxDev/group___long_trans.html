Long Transactions
ZRX Introductory Concepts » Database Operations

Long Transactions provide a way for ZRX Appplications to check out entities from a database and modify them in current database, then check them back.
The original database's objects are updated to what have be changed.
This example shows: how to change the color of the entities.

void CdatabaseOperations::longTrans()
{
AcDbDatabase *pSourceDb = new AcDbDatabase(Adesk::kFalse);
Acad::ErrorStatus es = Acad::eOk;
if(Acad::eOk != pSourceDb->readDwgFile(_T("Drawing1.dwg")))
{
acedAlert(_T("Readin the file fail."));
return ;
}
AcDbBlockTable* pBlockTable=NULL;
AcDbObjectId entId=AcDbObjectId::kNull;
AcDbBlockTableRecordIterator* blkIer;
AcDbObjectIdArray objectArr;
AcDbEntity* ent;
es=pSourceDb->getBlockTable(pBlockTable, AcDb::kForRead);
if (es==Acad::eOk)
{
AcDbBlockTableRecord* pBlockTableRecord=NULL;
es=pBlockTable->getAt(ACDB_MODEL_SPACE, pBlockTableRecord, AcDb::kForWrite);
pBlockTable->close();
if (es==Acad::eOk)
{
es=pBlockTableRecord->newIterator(blkIer);
if (es==Acad::eOk)
{
for (blkIer->start(); !blkIer->done(); blkIer->step())
{
if (blkIer->getEntity(ent, AcDb::kForRead)==Acad::eOk)
{
objectArr.append(ent->objectId());
}
ent->close();
}
}
delete blkIer;
}
pBlockTableRecord->close();
}
if (objectArr.isEmpty())
{
acedAlert(_T("Drawing1.dwg is empty."));
return ;
}
AcDbDatabase *pDb = acdbHostApplicationServices()->workingDatabase();
if (pDb != NULL)
{
AcDbBlockTable* pBlockTable=NULL;
AcDbObjectId entId=AcDbObjectId::kNull;
Acad::ErrorStatus es=Acad::eOk;
es=pDb->getBlockTable(pBlockTable, AcDb::kForRead);
if (es==Acad::eOk)
{
AcDbBlockTableRecord* pBlockTableRecord=NULL;
es=pBlockTable->getAt(ACDB_MODEL_SPACE, pBlockTableRecord, AcDb::kForRead);
pBlockTable->close();
if (es==Acad::eOk)
{
entId=pBlockTableRecord->objectId();
pBlockTableRecord->close();
AcDbIdMapping err;
AcDbObjectId checkoutId;
acapLongTransactionManagerPtr()->checkOut(checkoutId, objectArr, entId, err);
int cIndex;
if (acedGetInt(_T("Set Entity color Index :"), &cIndex)==RTNORM)
{
AcDbObject* pcheck;
if (acdbOpenObject(pcheck, checkoutId, AcDb::kForRead)==Acad::eOk)
{
AcDbLongTransaction* pLong=AcDbLongTransaction::cast(pcheck);
if (pLong!=NULL)
{
AcDbLongTransWorkSetIterator* pIter;
es=pLong->newWorkSetIterator(pIter);
if (es==Acad::eOk)
{
for (pIter->start(); !pIter->done(); pIter->step())
{
AcDbEntity* pEnt;
acdbOpenAcDbEntity(pEnt, pIter->objectId(), AcDb::kForWrite);
pEnt->setColorIndex(cIndex);
pEnt->close();
}
delete pIter;
}
}
pcheck->close();
}
}
acapLongTransactionManagerPtr()->checkIn(checkoutId, err);
pSourceDb->saveAs(_T("Drawing1.dwg"));
delete pSourceDb;
acedAlert(_T("check Drawing1.dwg file."));
}
}
}
}

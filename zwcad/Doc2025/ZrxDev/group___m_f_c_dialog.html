MFC model dialog
Basic Interaction with ZWCAD » MFC Support in ZRX

ZRX Application can directly use The Microsoft Foundation Class(MFC) libary. Developer can use MFC implement design standard user interfaces easily.
This example shows: how to build up a standard dialog. Click OK or Cancel button to close, and click "Button1" to show a welcome message.

class MFCDlg : public CDialog
{
public:
MFCDlg(CWnd* pParent = NULL);
enum { IDD = IDD_DIALOG1 };
protected:
virtual void DoDataExchange(CDataExchange* pDX);
DECLARE_MESSAGE_MAP()
public:
afx_msg void OnBnClickedButton1();
};
MFCDlg::MFCDlg(CWnd* pParent )
: CDialog(MFCDlg::IDD, pParent)
{
}
void MFCDlg::DoDataExchange(CDataExchange* pDX)
{
CDialog::DoDataExchange(pDX);
}
BEGIN_MESSAGE_MAP(MFCDlg, CDialog)
ON_BN_CLICKED(IDC_BUTTON1, &MFCDlg::OnBnClickedButton1)
END_MESSAGE_MAP()
void MFCDlg::OnBnClickedButton1()
{
MessageBox(_T("HelloWorld!"), _T("DlgSample"), MB_OK);
}
void CMFCSupport::MFCDialog()
{
MFCDlg mydialog;
mydialog.DoModal();
}

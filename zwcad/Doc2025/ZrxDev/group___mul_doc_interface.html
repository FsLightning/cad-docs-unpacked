The Multiple Document Interface
Advanced Topics

Platform support multiple document.
The graphic document is different from the graphics database.
The class corresponding to the graphics database is AcDbDatabase, and the class corresponding to the graphics document is AcApDocument.
Each open graph has an associated document object, the document object contains a lot of useful information,
such as file name, the CDocument object of MFC, the current graphics database, and the current file format.
Commands that work within the application context can create and switch documents, and commands that work within the document context can edit and modify the current document's graphics database.
This setion shows: how to have multiple drawings in a session.


When using "acedRegCmds->addCommand" to register a command, setting the parameter "ACRX_CMD_SESSION" to register commands running in the application context.

acedRegCmds->addCommand(_T("MDIDocument"),_T("CreateDraw"),_T("CreateDraw"),ACRX_CMD_SESSION, MDIDocument::CreateDraw,NULL,
-1,theArxDLL.ModuleResourceInstance());

void MDIDocument::CreateDraw(void)
{
Acad::ErrorStatus es;
es = acDocManager->lockDocument(acDocManager->curDocument());
int count=0;
AcDbDatabase *pDb = acdbHostApplicationServices()->workingDatabase();
if (pDb != NULL)
{
AcDbBlockTable* pBlockTable=NULL;
AcDbObjectId entId=AcDbObjectId::kNull;
Acad::ErrorStatus es=Acad::eOk;
es=pDb->getBlockTable(pBlockTable, AcDb::kForRead);
if (es==Acad::eOk)
{
AcDbBlockTableRecord* pBlockTableRecord=NULL;
es=pBlockTable->getAt(ACDB_MODEL_SPACE, pBlockTableRecord, AcDb::kForWrite);
pBlockTable->close();
if (es==Acad::eOk)
{
AcDbBlockTableRecordIterator* blkIer=NULL;
AcDbEntity* ent=NULL;
es=pBlockTableRecord->newIterator(blkIer);
if (es==Acad::eOk)
{
for (blkIer->start(); !blkIer->done(); blkIer->step())
{
if (blkIer->getEntity(ent, AcDb::kForRead)==Acad::eOk)
{
count++;
}
ent->close();
}
}
delete blkIer;
pBlockTableRecord->close();
}
}
}
acDocManager->unlockDocument(acDocManager->curDocument());
es = acDocManager->appContextNewDocument(NULL);
if (es == Acad::eOk)
{
es = acDocManager->lockDocument(acDocManager->curDocument());
//add
pDb = acdbHostApplicationServices()->workingDatabase();
if (pDb != NULL)
{
AcDbBlockTable* pBlockTable=NULL;
AcDbObjectId entId=AcDbObjectId::kNull;
Acad::ErrorStatus es=Acad::eOk;
es=pDb->getBlockTable(pBlockTable, AcDb::kForRead);
if (es==Acad::eOk)
{
AcDbBlockTableRecord* pBlockTableRecord=NULL;
es=pBlockTable->getAt(ACDB_MODEL_SPACE, pBlockTableRecord, AcDb::kForWrite);
pBlockTable->close();
if (es==Acad::eOk)
{
AcDbObjectId entId=AcDbObjectId::kNull;
for(int i=0; i<count; i++)
{
double y = i * 20;
AcGePoint3d startpoint(0, y, 0);
AcGePoint3d endpoint(100, y, 0);
AcDbLine* pLine=new AcDbLine(startpoint, endpoint);
es=pBlockTableRecord->appendZcDbEntity(entId, pLine);
if (es==Acad::eOk)
{
pLine->close();
}
}
pBlockTableRecord->close();
}
}
}
es = acDocManager->unlockDocument(acDocManager->curDocument());
}
}

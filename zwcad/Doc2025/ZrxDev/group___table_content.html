The following example demonstrates to create a table template use selected table.
Then newly created table will have the same structure, formatting, and overrides as the template.

void createTableTemplate()
{
Acad::ErrorStatus es;
AcDbTable *pTbl = NULL;
AcDbObjectId pEntId;
ads_name en;
ads_point pt;
if (acedEntSel(_T("\nPlease select a table for get table style: "), en, pt)
!= RTNORM)
{
acutPrintf(_T("Nothing Selected\n"));
return;
}
acdbGetObjectId(pEntId, en);
acdbOpenObject(pTbl, pEntId, AcDb::kForWrite);
if(NULL == pTbl)
{
acutPrintf(_T("\n Selected entity was not a table!"));
return;
}
static ACHAR sTableTemplate[MAX_PATH] = _T("TableTemplate");
static ACHAR sTableStyle[MAX_PATH] = _T("TableStyle");
AcDbTableTemplate* pTblTpl = new AcDbTableTemplate();
es = pTblTpl->capture(pTbl, AcDb::kTableCopySkipContent);
es = pTblTpl->setName(sTableTemplate);
AcDbTableStyle* pTblSty = new AcDbTableStyle();
AcDbDatabase *pDb = acdbHostApplicationServices()->workingDatabase();
assert(pDb);
AcDbDictionary *pDict = NULL;
es = acdbHostApplicationServices()->workingDatabase()->getTableStyleDictionary(pDict,AcDb::kForRead);
if(pDict->has(sTableStyle))
{
es = pDict->upgradeOpen();
es = pDict->remove(sTableStyle);
}
AcDbObjectId id1;
es = pDict->upgradeOpen();
if(Acad::eOk != (es = pDict->setAt(sTableStyle,pTblSty,id1)))
{
pDict->close();
delete pTblSty;
delete pTblTpl;
es = pTbl->close();
acutPrintf(_T("\nUnable to add new Table Style"));
return;
}
es = pDict->close();
assert(es == Acad::eOk);
AcDbObjectId id2;
es = pTblSty->setTemplate(pTblTpl, AcDb::kMergeCellStyleNone, id2);
if( es != Acad::eOk )
{
delete pTblTpl;
pTblSty->close();
pTbl->close();
acutPrintf(_T("\nError in setting data table template!"));
return;
}
es = pTblSty->close();
es = pTblTpl->close();
es = pTbl->close();
}

The following code adds a xrecord into a dictionary.

void ContainerObj::AddXrecord(AcDbObjectId dictId)
{
AcDbDictionary *pDict;
acdbOpenObject(pDict,dictId,AcDb::kForWrite);
AcDbObjectId xrecordId;
AcDbXrecord *pXrecord = new AcDbXrecord();
pDict->setAt(_T("Xrecord"),pXrecord,xrecordId);
struct resbuf *pRb;
pRb = acutBuildList(AcDb::kDxfText,
_T("Xrecord List"),
AcDb::kDxfInt32,10,
AcDb::kDxfReal,3.14,
RTNONE);
pXrecord->setFromRbChain(*pRb);
pXrecord->close();
pDict->close();
acutRelRb(pRb);
}

The following codes shows how to view the Xrecord;

void ContainerObj::ViewXrecord(AcDbObjectId dictId)
{
AcDbDictionary *pDict;
acdbOpenObject(pDict,dictId,AcDb::kForRead);
AcDbXrecord *pXrecord;
pDict->getAt(_T("Xrecord"),(AcDbObject* &)pXrecord,AcDb::kForRead);
struct resbuf *pRb;
pXrecord->rbChain(&pRb);
if(pRb!=NULL)
{
struct resbuf *pTemp;
pTemp = pRb;
acutPrintf(_T("\n %s"),pTemp->resval.rstring);
pTemp = pTemp->rbnext;
acutPrintf(_T("\n %d"),pTemp->resval.rint);
pTemp = pTemp->rbnext;
acutPrintf(_T("\n %f"),pTemp->resval.rreal);
}
pXrecord->close();
pDict->close();
acutRelRb(pRb);
}
